# Filenames of the VideoCore GPU firmware files
start_file=start4.elf
fixup_file=fixup4.dat

# Filename of the kernel image on the boot partition
kernel=Image

# Start kernel in 64 bit mode
arm_64bit=1

# Disable overscan assuming the display supports displaying the full resolution
disable_overscan=1

# How much memory in MB to assign to the GPU on Pi models having
# 256, 512 or 1024 MB total memory
gpu_mem_256=100
gpu_mem_512=100
gpu_mem_1024=100

# Enable UART0 for serial console on ttyAMA0
dtoverlay=miniuart-bt

# Import custom overlay that does nothing
dtoverlay=custom-do-nothing
